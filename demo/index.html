Get Started

Installation

1. Install the package

npm install tail-vapor

2. Add to tailwind.config.js

module.exports = {
  plugins: [
    require('tail-vapor'),
  ],
}

3. Import styles (optional)

@import 'tail-vapor/src/styles.css';

4. Apply a theme

<body data-theme="neon-city">
  <!-- Your content -->
</body>

<!-- Available themes: -->
<!-- sunset, neon-city, miami-vice, digital-dreams, midnight-chrome -->
<!-- arcade, outrun, cyber-noir -->
